function runTimer(){
  clearInterval(ws.iv);
  const e=EX[ws.idx], C=427.25;
  function tick(){
    if(ws.paused) return;
    document.getElementById('w-ring').style.strokeDashoffset=C*(1-ws.rem/e.dur);
    const m=Math.floor(ws.rem/60),s=ws.rem%60;
    document.getElementById('w-timer').textContent=`${m}:${s.toString().padStart(2,'0')}`;
    document.getElementById('w-tlabel').textContent=ws.rem>0?'remaining':'done!';
    updCue(e,ws.rem);
    const done=EX.slice(0,ws.idx).reduce((a,x)=>a+x.dur,0)+(e.dur-ws.rem);
    document.getElementById('wt-bar').style.width=Math.min((done/TOTAL)*100,100)+'%';
    const el=Math.floor((Date.now()-ws.start)/1000);
    document.getElementById('wt-total').textContent=`${Math.floor(el/60)}:${(el%60).toString().padStart(2,'0')} / ${Math.floor(TOTAL/60)}:${(TOTAL%60).toString().padStart(2,'0')}`;
    if(ws.rem<=0){
      clearInterval(ws.iv);
      if(ws.idx<EX.length-1) setTimeout(()=>{if(!ws.paused)nextEx();},1500);
      else setTimeout(doneW,1500);
    } else ws.rem--;
  }
  tick(); ws.iv=setInterval(tick,1000);
}

function toggleDay(i){
  if(ap.days.includes(i)){if(ap.days.length>1)ap.days=ap.days.filter(d=>d!==i);}
  else{if(ap.days.length<2){ap.days.push(i);ap.days.sort((a,b)=>a-b);}else{ap.days.shift();ap.days.push(i);ap.days.sort((a,b)=>a-b);}}
  const d=document.getElementById('day-warn');
  if(ap.days.length===2){const diff=Math.abs(ap.days[1]-ap.days[0]);d.style.display=Math.min(diff,7-diff)<2?'block':'none';}
  else d.style.display='none';
  renderDays(); renderWeek(); save();
}

      <div style="display:flex;gap:7px;flex:1">

function toggleSess(w,s){
  const keys=Object.keys(ap.tracker).sort(),idx=w*2+s;
  if(idx<keys.length) delete ap.tracker[keys[idx]];
  else if(idx===keys.length) ap.tracker[new Date().toISOString().split('T')[0]+'_'+idx]=true;
  save(); renderTr();
}

}
